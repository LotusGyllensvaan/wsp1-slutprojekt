def self.products_in_cart(shopping_session_id)
  db.execute('
    SELECT p.id AS product_id, p.name, p.description, p.SKU, p.category, p.price, ci.quantity
    FROM cart_item ci
    JOIN product p ON ci.product_id = p.id
    WHERE ci.session_id = ?',
    [shopping_session_id]
  )
end
